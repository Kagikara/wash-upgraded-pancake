// 1) 拆股/送转导致跳空
class SplitDividendGapRule implements Rule:
    name = "SplitDividendGapRule"
    category = "CorporateActions"

    function validate(records, context):
        issues = []
        grouped = group_by_ticker(records)
        for ticker, rows in grouped:
            sorted_rows = sort_by_date(rows)
            for i from 1 to len(sorted_rows)-1:
                prev = sorted_rows[i-1]
                cur  = sorted_rows[i]
                if is_large_gap(prev.close, cur.open):
                    if context.corp_actions.has_split_or_bonus(ticker, cur.date):
                        issues.append(Issue(
                            type="SPLIT_BONUS_GAP",
                            category=category,
                            ticker=ticker,
                            date=cur.date,
                            field="open",
                            value=cur.open,
                            detail="Gap explained by split/bonus action"
                        ))
                    else:
                        issues.append(Issue(
                            type="UNEXPLAINED_GAP",
                            category=category,
                            ticker=ticker,
                            date=cur.date,
                            field="open",
                            value=cur.open,
                            detail="Large gap without split/bonus action"
                        ))
        return issues
