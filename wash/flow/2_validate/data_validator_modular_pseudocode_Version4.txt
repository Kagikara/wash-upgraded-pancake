// 1) 涨跌停板限制（可交易性问题）
class LimitUpDownRule implements Rule:
    name = "LimitUpDownRule"
    category = "Microstructure"

    function validate(records, context):
        issues = []
        grouped = group_by_ticker(records)
        for ticker, rows in grouped:
            sorted_rows = sort_by_date(rows)
            prev_close = None
            for row in sorted_rows:
                if prev_close is not None:
                    limit = context.market_rules.price_limit(ticker, row.date) // 10% or 5% for ST
                    if abs(row.close - prev_close) / prev_close > limit:
                        issues.append(Issue(
                            type="PRICE_LIMIT_EXCEEDED",
                            category=category,
                            ticker=ticker,
                            date=row.date,
                            field="close",
                            value=row.close,
                            detail="Close exceeds price limit rule"
                        ))
                prev_close = row.close
        return issues


// 2) 成交量为 0 但价格发生位移
class ZeroVolumePriceMoveRule implements Rule:
    name = "ZeroVolumePriceMoveRule"
    category = "Microstructure"

    function validate(records, context):
        issues = []
        grouped = group_by_ticker(records)
        for ticker, rows in grouped:
            sorted_rows = sort_by_date(rows)
            prev_close = None
            for row in sorted_rows:
                if prev_close is not None and row.volume == 0:
                    if row.open != prev_close or row.close != prev_close or
                       row.high != prev_close or row.low != prev_close:
                        issues.append(Issue(
                            type="VOLUME_ZERO_PRICE_MOVED",
                            category=category,
                            ticker=ticker,
                            date=row.date,
                            field="volume",
                            value=row.volume,
                            detail="Price moved despite zero volume"
                        ))
                prev_close = row.close
        return issues

// 5) 流动性枯竭（价差极宽导致噪音）
class LiquidityDryUpRule implements Rule:
    name = "LiquidityDryUpRule"
    category = "Microstructure"

    function validate(records, context):
        issues = []
        for row in records:
            spread = compute_spread(row) // 需 L2 或盘口数据
            if spread is not None and spread > context.market_rules.max_spread_threshold:
                issues.append(Issue(
                    type="LIQUIDITY_DRY_UP",
                    category=category,
                    ticker=row.ticker,
                    date=row.date,
                    field="spread",
                    value=spread,
                    detail="Spread too wide, poor tradability"
                ))
        return issues
